import java.util.Scanner;
class factorial{
	public static void main(String args[]){
		Scanner scan=new Scanner();
		System.out.print("please enter the nth number ");
		int n=scan.nextInt();
		int i=0;
		int a=0;
		int b=0;
		while (n>i){
			a,b=b,a+b;
			i+=1
			}
		System.out.println("the factorial of "+n+" is "+b);
		}
	}
